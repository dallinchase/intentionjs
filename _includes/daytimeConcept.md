var time = intent.responsive({
   ID: 'time',
   contexts: [
      {name:'night',min:20},
      {name:'evening',min:17},
      {name:'sunset',min:16},
      {name:'day',min:9},
      {name:'morning',min:0}
   ],
   matcher: function(test, ctx) {
      return test >= ctx.min;
   },
   measure: function(arg) {
      var time = new Date();
      return time.getHours();
   }
});
time.respond(); //check the current context
